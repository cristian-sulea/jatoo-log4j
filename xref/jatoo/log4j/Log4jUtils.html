18  package jatoo.log4j;
19  
20  import java.io.File;
21  import java.io.FilenameFilter;
22  import java.io.IOException;
23  import java.io.InputStream;
24  import java.io.InterruptedIOException;
25  import java.net.MalformedURLException;
26  import java.net.URL;
27  import java.net.URLConnection;
28  import java.util.Enumeration;
29  import java.util.Properties;
30  
31  import org.apache.log4j.LogManager;
32  import org.apache.log4j.PropertyConfigurator;
33  import org.apache.log4j.helpers.LogLog;
34  
35  /**
36   * A collection of utility methods to ease the work with Apache Log4jâ„¢ logging
37   * library.
38   * 
39   * @author <a href="http://cristian.sulea.net" rel="author">Cristian Sulea</a>
40   * @version 4.4, September 14, 2015
41   */
42  public final class Log4jUtils {
43  
44    /** The logs folder parameter used in configuration files. */
45    public static final String SYSTEM_PROPERTY_LOGS_FOLDER = "logs.folder";
46  
47    /** Be sure that the init/destroy methods are executed only once. */
48    private static boolean isInitialized = false;
49  
50    /**
51     * Utility classes should not have a public or default constructor.
52     */
53    private Log4jUtils() {}
54  
55    /**
56     * Initializes Log4J with the <code>logs</code> folder created in the working
57     * directory (where the JVM is started).
58     */
59    public static void init() {
60      init(new File(System.getProperty("user.dir")));
61    }
62  
63    /**
64     * Initializes Log4J with the <code>logs</code> folder created in the
65     * specified folder.
66     * 
67     * @param logsFolder
68     *          the parent folder of the <code>logs</code> folder
69     */
70    public static void init(final File logsFolder) {
71      init(logsFolder, new File(System.getProperty("user.dir")));
72    }
73  
74    /**
75     * Initializes Log4J with the <code>logs</code> folder created in the
76     * specified folder, trying to search for new properties or loggers in the
77     * second specified folder.
78     * 
79     * @param logsFolder
80     *          the parent folder of the <code>logs</code> folder
81     * @param configFolder
82     *          the parent folder of the <code>config</code> folder
83     */
84    public static void init(final File logsFolder, final File configFolder) {
85  
86      if (!isInitialized) {
87        synchronized (Log4jUtils.class) {
88          if (!isInitialized) {
89  
90            //
91            // ensure logs folder
92            // and add him to the system properties
93  
94            File logsFolderFile = new File(logsFolder, "logs");
95            logsFolderFile.mkdirs();
96  
97            System.setProperty(SYSTEM_PROPERTY_LOGS_FOLDER, logsFolderFile.getAbsolutePath());
98  
99            Properties properties = new Properties();
100 
101           //
102           // configure from jar
103 
104           updatePropertiesFromURL(properties, Log4jUtils.class.getClassLoader().getResource("META-INF/log4j/log4j.properties"), true);
105 
106           //
107           // configure from config folder
108 
109           updatePropertiesFromFile(properties, configFolder.getAbsolutePath() + "/config/log4j/log4j.properties", false);
110 
111           //
112           // load loggers from jar
113 
114           try {
115 
116             Enumeration<URL> loggersEnumeration = Log4jUtils.class.getClassLoader().getResources("META-INF/log4j/loggers.properties");
117 
118             while (loggersEnumeration.hasMoreElements()) {
119               updatePropertiesFromURL(properties, loggersEnumeration.nextElement(), true);
120             }
121           }
122 
123           catch (Exception e) {
124             LogLog.error("Could not load loggers from jar(s) [META-INF/log4j/loggers.properties].", e);
125           }
126 
127           //
128           // load loggers from config folder
129 
130           String[] loggers = new File(configFolder, "config/log4j/loggers/").list(new FilenameFilter() {
131             public boolean accept(final File dir, final String name) {
132               return name.endsWith(".properties");
133             }
134           });
135 
136           if (loggers != null && loggers.length > 0) {
137             for (String logger : loggers) {
138               updatePropertiesFromFile(properties, configFolder.getAbsolutePath() + "/config/log4j/loggers/" + logger, false);
139             }
140           }
141 
142           //
143           // configure
144 
145           PropertyConfigurator.configure(properties);
146 
147           //
148           // done
149 
150           isInitialized = true;
151         }
152       }
153     }
154   }
155 
156   /**
157    * @return <code>true</code> if the <code>Log4J</code> is initialized;
158    *         <code>false</code> otherwise
159    */
160   public static boolean isInitialized() {
161     synchronized (Log4jUtils.class) {
162       return isInitialized;
163     }
164   }
165 
166   /**
167    * Release all the resources.
168    */
169   public static void destroy() {
170 
171     if (isInitialized) {
172       synchronized (Log4jUtils.class) {
173         if (isInitialized) {
174 
175           System.clearProperty(SYSTEM_PROPERTY_LOGS_FOLDER);
176 
177           LogManager.shutdown();
178 
179           isInitialized = false;
180         }
181       }
182     }
183   }
184 
185   /**
186    * Update the {@link Properties} with values from the specified {@link URL}.
187    * 
188    * @param properties
189    *          the {@link Properties} object to be updated
190    * @param url
191    *          the configuration file with the new values
192    * @param logError
193    *          <code>true</code> if an error must be logged when something goes
194    *          wrong, like file missing for example
195    */
196   private static void updatePropertiesFromURL(final Properties properties, final URL url, final boolean logError) {
197 
198     LogLog.debug("Reading configuration file... [" + url + "]");
199 
200     //
201     // read properties from provided URL
202 
203     Properties p = new Properties();
204 
205     InputStream stream = null;
206     URLConnection connection = null;
207 
208     try {
209 
210       connection = url.openConnection();
211       connection.setUseCaches(false);
212 
213       stream = connection.getInputStream();
214 
215       p.load(stream);
216     }
217 
218     catch (Exception e) {
219 
220       if (e instanceof InterruptedIOException || e instanceof InterruptedException) {
221         Thread.currentThread().interrupt();
222       }
223 
224       if (logError) {
225         LogLog.error("Could not read configuration file... [" + url + "]", e);
226         LogLog.error("Ignoring configuration file... [" + url + "]");
227       }
228     }
229 
230     finally {
231       if (stream != null) {
232 
233         try {
234           stream.close();
235         }
236 
237         catch (InterruptedIOException e) {
238           Thread.currentThread().interrupt();
239           if (logError) {
240             LogLog.error("Hmm... In this case (InterruptedIOException) call also #interrupt() on the current thread! [" + url + "]", e);
241           }
242         }
243 
244         catch (IOException e) {
245           if (logError) {
246             LogLog.error("The stream.close() call failed. [" + url + "]", e);
247           }
248         }
249 
250         catch (RuntimeException e) {
251           if (logError) {
252             LogLog.error("This is strange... How can i have something else (except an io exception) on a stream.close() call? [" + url + "]", e);
253           }
254         }
255       }
256     }
257 
258     //
259     // update properties
260 
261     for (String key : p.stringPropertyNames()) {
262       properties.setProperty(key, p.getProperty(key));
263     }
264   }
265 
266   /**
267    * Update the {@link Properties} with values from the specified file.
268    * 
269    * @param properties
270    *          the {@link Properties} object to be updated
271    * @param file
272    *          the configuration file with the new values
273    * @param logError
274    *          <code>true</code> if an error must be logged when something goes
275    *          wrong, like file missing for example
276    */
277   private static void updatePropertiesFromFile(final Properties properties, final String file, final boolean logError) {
278 
279     URL url;
280 
281     try {
282       url = new File(file).toURI().toURL();
283     }
284 
285     catch (MalformedURLException e) {
286 
287       LogLog.error("Could not convert file to URL [" + file + "].", e);
288       LogLog.error("Ignoring configuration file [" + file + "].");
289 
290       return;
291     }
292 
293     updatePropertiesFromURL(properties, url, logError);
294   }
